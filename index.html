function numberOfPairs(gloves) {
  let colors = new Set(gloves);
  let resultSum = 0;
  for (let color of colors) {
    let colorGloves = gloves.filter((glove) => glove === color);
    let oneColorPairAmount = Math.floor(colorGloves.length / 2);
    resultSum += oneColorPairAmount;
  }
  return resultSum;
}
